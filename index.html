Leaderboards, achievements, user levels,
dynamic content, in-game newsletter subscriptions

Open source. Scalable. Free.

Developers, take control of your backend

Playtomic.org is the new home of the tools once provided by Playtomic - achievements, leaderboards, player levels, gamevars, geoip and in-game newsletter subscription - as a simple to host, open source platform for your game compatible with most gaming platforms.

Setting up your server is very easy and for small games will probably stay under the free limits of the selected hosting providers.

It's your game, now it's your backend too.

Game APIs

Playtomic has extensive documentation, compatibility and tests/examples for all major gaming platforms.

  • ActionScript 3
  • Android / Java
  • Haxe / OpenFL
  • HTML5 / JavaScript
  • iOS / Objective C
  • Mono and .NET
  • Unity3d

Achievements

Cross-platform, versatile and flexible achievements system.

Features

  • Award once, overwrite, or allowduplicates
  • Filter by device or website
  • Filter by custom fields like difficulty to differentiate between different levels of achievement
  • Include own and friends' status when retrieving lists
  • Filter by own and friends' ids to make everything social
  • View awarded achievements as a stream of updates
  • Group the achievement stream to show only the most recent for each player

Achievements example

// let's start with listing
var options = {
  playerid: "my_id" // optional from facebook or google or twitter or apple or whatever
}

Playtomic.Achievements.list(options, function(achievements, response){

    if(!response.success) {
      return console.log("something went wrong: " + response.errormessage);
    }

   /*
    * now we have our achievement list, it will be something like this:
    * [
    *  {
    *    achievement: "did something awesome"
    *  },
    *  {
    *    achievement: "killed the boss",
    *    player: {  // when the specified player has the achievement we have a 'player' object
    *               // with the data, when we include our friends we get an
    *               // array called 'friends' of objects the same as this
    *      playerid: "my_id",
    *      playername: "fred",
    *      date: new Date(),
    *      rdate: "3 days ago",
    *      fields: {
    *        difficulty: "easy"
    *      }
    *    }
    *  }
    * ]
    */

  // now let's make it more interesting by including our friends
  options.friendslist = [ "an", "array", "of", "friends", "ids"];

  Playtomic.Achievements.list(options, function(achievements, response){
      if(!response.success) {
        return console.log("something went wrong: " + response.errormessage);
      }

    for(var i=0; i<achievements.length; i++) {
      var message = achievements[i].achievement;

      if(achievements[i].player) {
        message += ", awarded to player on " + achievements[i].player.date;
      }

      if(achievements[i].friends) {

        var friends = "";

        if(achievements[i].friends.length > 3) {
          friends = achievements[i].friends[0].playername + ", " +
                    achievements[i].friends[1].playername + ", " +
                    achievements[i].friends[2].playername + " and " +
                    (achievements[i].friends.length - 3) + " more";
        } else {
          for(var j=0; j<achievements[i].friends.length; j++) {
            friends += (j > 0 ? "," : "") + achievements[i].friends[j].playername;
          }
        }

        message += ", awarded to " + achievements[i].friends.length + " friends: " + friends;
      }

      console.log(message);
      // output: killed the boss, awarded to player on May 10 2013,
      // awarded to 7 friends: tom, dick, harry and 4 more
    }

    // let's view the achievements chronologically by me
    options = {
      playerid: "my_id",
    }

    Playtomic.Achievements.stream(options, function(achievements, numachievements, response) {
        if(!response.success) {
          return console.log("something went wrong: " + response.errormessage);
        }

      console.log("I have " + numachievements + " achievements,
                  here are the latest");

      for(var i=0; i<achievements.length; i++) {
        console.log(achievements[i].awarded.achievement + " on " + achievements[i].date +
                    " (" + achievements[i].fields.difficulty + ")");
      }

      // and now let's view me and my friends with one-update-per-person
      options.friendslist =  ["2", "3", "4", "5"];
      options.group = true; // this triggers making it the most recent for each person

      Playtomic.Achievements.stream(options, function(achievements, numachievements, response) {
        if(!response.success) {
          return console.log("something went wrong: " + response.errormessage);
        }

        console.log("My friends and I have " + numachievements + " achievements, here are the latest");

        for(var i=0; i<achievements.length; i++) {
          console.log(achievements[i].awarded.achievement + " to " + achievements[i].playername +
		              " on " + achievements[i].date + " (" + achievements[i].fields.difficulty + ")";
          // output: killed the boss, awarded to ben on May 14 2013 (easy)
          // output: killed the boss, awarded to tom on May 10 2013 (hard)
        }

        // and now let's make it just the hard achievements
        options.filter = {
          difficulty: "hard"
        }

        Playtomic.Achievements.stream(options, function(achievements, numachievements, response) {
            if(!response.success) {
              return console.log("something went wrong: " + response.errormessage);
            }

          console.log("My friends and I have " + numachievements + " hard achievements, here are the latest");

          for(var i=0; i<achievements.length; i++) {
            console.log(achievements[i].awarded.achievement + " to " + achievements[i].playername +
                        " on " + achievements[i].date +
                        " (" + achievements[i].fields.difficulty + ")");
            // output: killed the boss, awarded to tom on May 10 2013 (hard)
          }
        });
      });
    });
  });
});

Newsletters

Build your mailing list in-game

Features

  • Built on top of the very powerful MailChimp API
  • Players don't have to leave your game
  • Free for up to 2,000 players with 12,000 emails per month
  • Pay as you go and subscription options available

On the server side you can turn double opt-in off and disable updating a player's records if they re-submit their information.

Leaderboards

Incredibly rich capabilities available to build on.

Features

  • Highest or lowest
  • Custom fields for additional data like characters, kills, or anything else you want
  • Filtering by custom fields to automatically create sub-leaderboards on that data
  • Five simple modes: TODAY, LAST7DAYS, LAST30DAYS, ALLTIME, NEWEST (a score stream)
  • Filter by friends' ids to make your leaderboards social
  • Filter by player's id to make your leaderboards personal
  • Combine all of the above to make the most powerful leaderboards around!

Leaderboard example

Imagine a racing game that you and your friends play. Let's walk through some of the awesome capabilities waiting for you in these leaderboards using the HTML5 API.

// we just tore up race #10 in our lambo
var score = {
  playername: "fred",
  playerid: "my_id", // optional facebook, apple, google, device, whatever you want
  points: 243, // seconds
  allowduplicates: false, // only save the player's best score
  lowest: true, // low score is best
  fields: {  // save this additional data with the score
    car: "Lamborghini Gallardo",
    race: 10
  },
  filters: {  // return scores with these fields
    race: 10
  }
}

// by using saveAndList we submit a score and return the 'page' of scores the
// ranked score sits on, eg 17th score on 10 per page will give us scores #11 - 20
Playtomic.Leaderboards.saveAndList(score, function(scores, numscores, response) {

  if(!response.success) {
    return console.log("something went wrong: " + response.errormessage);
  }

  console.log("Best times for race #10 (" + numscores + " total scores)");
  var i;

  for(i=0; i<scores.length; i++) {
    console.log(scores[i].rank + "   " + scores[i].points + "   " +
                scores[i].playername + " driving " + scores[i].fields.car);
  }

  // let's drill down further and just look at that car
  var list = {
    lowest: true,
    perpage: 10,
    filters: {
      car: "Lamborghini Gallardo",
      race: 10
    }
  }

  Playtomic.Leaderboards.list(list, function(scores, numscores, response) {

    if(!response.success) {
      return console.log("something went wrong: " + response.errormessage);
    }

    console.log("Best times for Lamborghini Gallardos on race #10");

    for(i=0; i<scores.length; i++) {
      console.log(scores[i].rank + "   " + scores[i].points + "   " + scores[i].playername);
    }

    // let's drill down further and make this about my personal bests
    var myleaderboard = {
      lowest: true,
      perpage: 10,
      playerid: "my_id",
      filters: {
        car: "Lamborghini Gallardo",
        race: 10
      }
    }

    Playtomic.Leaderboards.list(myleaderboard, function(scores, numscores, response) {

      if(!response.success) {
        return console.log("something went wrong: " + response.errormessage);
      }

      console.log("My best times for Lamborghini Gallardos on race #10");

      for(i=0; i<scores.length; i++) {
        console.log(scores[i].rank + "   " + scores[i].points + "   " + scores[i].rdate);
        // 'rdate' is a friendly date format relative to utc, like 7 minutes ago
      }

      // let's see what my friends are submitting as a stream
      var friendsactivity = {
        perpage: 20,
        friendslist: ["array of friends' playerids"],
        mode: "newest" // in 'newest' we get the scores as an unranked stream
      }

      Playtomic.Leaderboards.list(friendsactivity, function(scores, numscores, response) {

        if(!response.success) {
          return console.log("something went wrong: " + response.errormessage);
        }

        console.log("Latest scores from my friends");

        for(i=0; i<scores.length; i++) {
          console.log(scores[i].points + "   " + scores[i].playername +
                      " driving " + scores[i].fields.car + " on race " + scores[i].fields.race);
        }
      });
    });
  });
});

PlayerLevels

Cross-platform user created content.

Features

  • Rate creations
  • Custom fields for any meta data you want to include
  • Filtering by custom fields
  • Two simple modes: POPULAR and NEWEST
  • Fetch lists of levels with heavy data payloads excluded to reduce bandwidth
  • Filter by friends' ids to share content within a player's personal network
  • Filter by player's id to maintain a catalog of the player's creations
  • Create content on tablets or web to share with smaller form factors

PlayerLevel example

You've created a game where players can design their own levels for other people to play. This is how you'd do it with the HTML5 API.

// let's save our creation
var level = {
  name: "my level",
  playername: "ben",
  playerid: "my_id", // optional facebook, apple, google, device, whatever you want
  data: "this is the level data",
  fields: {
    shots: 9,
    enemies: 8
  }
}

Playtomic.PlayerLevels.save(level, function(l, response) {

  if(!response.success) {
    return console.log("something went wrong: " + response.errormessage);
  }

  console.log("my level has been saved with level id " + l.levelid);

  // someone wants to play the levels other people have made
  var options = {
    mode: "popular",
    perpage: 10,
    data: false // if your level data is big and you are retrieving lots of
                // levels at once you can exclude the data and just load it
                // individually 
  }

  Playtomic.PlayerLevels.list(options, function(levels, numlevels, response) {
    if(!response.success) {
      return console.log("something went wrong: " + response.errormessage);
    }

    console.log("loaded " + levels.length + " levels out of " + numlevels + " total");

    for(var i=0; i<levels.length; i++) {
      console.log(levels[i].name + " by " + levels[i].ben + ": " + levels[i].rating);
    }

    // now let's pretend we clicked on one of those levels to play it, because
    // we excluded the data when we fetched the list we now need to load it in
    // a seperate request
    Playtomic.PlayerLevels.load(selected.levelid, function(level, response) {

        if(!response.success) {
          return console.log("something went wrong: " + response.errormessage);
        }

      console.log("ready to play " + level.name);

      // so we beat this level and now we want to rate it out of 10
      Playtomic.PlayerLevels.rate(level.levelid, 8, function(response) {

          if(!response.success) {
            return console.log("something went wrong: " + response.errormessage);
          }

        console.log("rated " + level.levelid + " successfully, which increases " +
                    "the votes by 1 and the score by 8, from which the rating is calculated.");

          // now we want to play our friends' levels
          var friendslevels =  {
            mode: "popular",
            perpage: 10,
            data: true,
            friendslist: ["my", "friends", "playerids"]
          }

          Playtomic.PlayerLevels.list(friendslists, function(levels, numlevels, response) {

              if(!response.success) {
                return console.log("something went wrong: " + response.errormessage);
              }

              console.log("loaded " + levels.length + " levels out of " + numlevels + " total");

              for(var i=0; i<levels.length; i++) {
                console.log(levels[i].name + " by " + levels[i].ben + ": " + levels[i].rating);
              }

              // now i want to show off my own creations
              var mylevels = {
                mode: "popular",
                playerid: ""my_id"
              }

              Playtomic.PlayerLevels.list(mylevels, function(levels, numlevels, response) {

                if(!response.success) {
                  return console.log("something went wrong: " + response.errormessage);
                }

                console.log("loaded " + levels.length + " levels out of " + numlevels + " total");

                for(var i=0; i<levels.length; i++) {
                  console.log(levels[i].name + " by " + levels[i].ben + ": " + levels[i].rating);
                }

            });
          });
        });
      });
    });
  });
});

GameVars

Update your game without republishing

Features

  • Update at any time in your database
  • Tweak difficulty balancing
  • Store game configuration
  • Manage cross-promotions
  • Rotate between advertising partners

GameVars example

In our game we are going to store the entire configuration for our enemies. This data weighs in at 47 kilobytes so we are also going to store a 'version' we can cache to skip reloading the same data.

// first let's set up our default configuration for when the player is disconnected
var config = {
	version: 1,
	enemies: { // the enemy configuration
		badguy1: {
			hitpoints: 100,
			gun: 3,
			speed: 8,
			damage: 3
		},
		....
		badguy9000: {
			hitpoints: 10000000,
			gun: 12,
			speed: 47,
			damage: 120
		}
	},
	player: { // the player's configuration
		hitpoints: 1000,
		xplevels: [ // when to upgrade the player's level
			1000,
			....
			1024000
		]
	},
	powerups: { // probability of dropping cool stuff
		powerup1: 0.3
		....
		powerup10: 0.0005
	},
	crosspromotion: {  // will we promote our other game
		enabled: true,
		url: "https://itunes.apple.com/....",
		image: "https://the_url_to_my_image.png"
	}
}

// so now when the player opens up your game we load the 'version'
Playtomic.GameVars.loadSingle("version", function(gamevars, response)) {

  if(!response.success) {
    return console.log("something went wrong: " + response.errormessage);
  }

  // check if we need to make the bigger request for all of the data
  if(gamevars.version > config.version) {

    Playtomic.GameVars.load(function(gamevars, response) {

      if(!response.success) {
        return console.log("something went wrong: " + response.errormessage);
      }

      // and update our local config, you could do this in a variety of ways depending on
      // if you are storing one giant GameVar with the entire config:
      config = gamevars;

      // or doing each section as a separate GameVar
      config.version = gamevars.version;
      config.enemies = gamevars.enemies;
      config.player = gamevars.player;
      config.powerups = gamevars.powerups;
      config.crosspromotion = gamevars.crosspromotion;

      // or going even more granular
      config.enemies.badguy1 = gamevars.enemies_badguy1;
    });
  });
});
